## Install

```bash
npm install --save sharki
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'sharki'
import 'sharki/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
